vi_stress_threshold.2010 = { # Lose Humanity and Compassionate
	type = character_event
	title = vi_stress_threshold.2010.t
	desc = vi_stress_threshold.2010.desc

	theme = mental_break

	left_portrait = {
		character = root
		animation = rage
	}

	trigger = {
		stress_level >= 2
		NOR = {
			has_trait = sadistic
			has_character_flag = had_vi_stress_threshold2010
		}
	}

	weight_multiplier = {
		base = 1

		modifier = {
			add = 1
			has_trait = compassionate
		}
		modifier = {
			subtract = 0.75
			has_trait = callous
		}
	}

	immediate = {
		root = {
			add_character_modifier = {
				modifier = vi_vampire_madness_modifier
				years = 20
			}

			add_character_flag = {
				flag = had_vi_stress_threshold2010
				years = 10
			}
		}
	}

	option = {
		name = vi_stress_threshold.2010.a
		trigger = {
			has_trait = compassionate
		}
		remove_trait = compassionate
	}

	option = {
		name = vi_stress_threshold.2010.b
		trigger = {
			NOR = {
				has_trait = compassionate
				has_trait = callous
				has_trait = sadistic
			}
		}
		add_trait = callous
	}

	option = {
		name = vi_stress_threshold.2010.c
		trigger = {
			NOR = {
				has_trait = compassionate
				has_trait = sadistic
			}
		}
		if = {
			limit = {
				has_trait = callous
			}
			remove_trait = callous
		}
		add_trait = sadistic
	}

	option = {
		name = vi_stress_threshold.2010.d
	}
}
