	title = rvi_legacies.1003.t

			triggered_desc = {
				trigger = {
					scope:vi_traitsuck = {
						has_trait = physique_good_3
					}
				}

			}
			triggered_desc = {
				trigger = {
					scope:vi_traitsuck = {

			}

			if = {
				limit = { has_trait = physique_good_2 }
				remove_trait = physique_good_2
				add_trait = physique_good_3
			}
			else_if = {
				limit = { has_trait = physique_good_1 }
