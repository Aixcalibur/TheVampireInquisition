vi_vampire_add_perk_effect = {
	if = {
		limit = {
			has_perk = $PERK$
		}

		add_vi_vampire_lifestyle_xp = 800

	}
	else = {
		add_perk = $PERK$
	}
}

		}
	}
	change_current_weight = -25
}
