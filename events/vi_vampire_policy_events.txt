namespace = vi_vampire_policy

vi_vampire_policy.001 = {
	type = character_event
	title = vi_vampire_policy.001.t
	desc = vi_vampire_policy.001.desc

	theme = realm

	right_portrait = {
		character = scope:vi_councillor
		# animation = 
	}

	left_portrait = {
		character = root
		animation = personality_rational
	}

	override_background = {
		reference = vi_council_chamber_night
	}

	immediate = {

		add_character_flag = {
			flag = vi_vampire_policy_decided
			days = 1800
		}

		if = {
			limit = {
				employs_court_position = vi_castellan_court_position
			}
			random_courtier = {
				limit = {
					has_court_position = vi_castellan_court_position
				}
				save_scope_as = vi_councillor
			}
		}
		else_if = {
			limit = {
				any_councillor = {
					has_council_position = councillor_steward
				}
			}

			random_councillor = {
				limit = {
					has_council_position = councillor_steward
				}
				save_scope_as = vi_councillor
			}
		}
		else = {
			random_councillor = {
				save_scope_as = vi_councillor
			}
		}
		
		vi_vampire_policy_start_effect = yes
		
		root.primary_title.title_capital_county = {
			save_scope_as = lair_location
		}

		scope:lair_location.holder = {
			save_scope_as = vi_host
		}
	}


	option = {
		name = vi_vampire_policy.001.a

		ai_chance = {
			base = 50

			ai_value_modifier = {
				ai_compassion = 0.25
				ai_rationality = 1
				ai_greed = 0.25
			}
		}
		vi_vampire_county_normal_effect = yes
	}

	option = {

		name = vi_vampire_policy.001.b

		ai_chance = {
			base = 10

			modifier = {
				add = 20
				has_trait = craven
			}

			modifier = {
				add = 10
				gold > major_gold_value
			}

			modifier = {
				add = 50
				has_legitimacy = yes
				legitimacy_level = 2
			}

			modifier = {
				add = 100
				has_legitimacy = yes
				legitimacy_level <= 1
			}

			ai_value_modifier = {
				ai_compassion = 0.1
			}
		}

		vi_vampire_county_discreet_effect = yes
	}

	option = {
		name = vi_vampire_policy.001.c

		ai_chance = {
			base = 10

			modifier = {
				add = 100
				gold <= 0
			}

			opinion_modifier = {
				opinion_target = scope:vi_host
				multiplier = -1
			}

			modifier = {
				add = -50
				has_legitimacy = yes
				legitimacy_level = 2
			}

			modifier = {
				add = -100
				has_legitimacy = yes
				legitimacy_level <= 1
			}

			ai_value_modifier = {
				ai_compassion = -0.5 
				ai_boldness = 0.5
				ai_greed = 0.5
				ai_rationality = -0.25
			}
		}

		vi_vampire_county_drain_effect = yes

	}

	option = {
		name = vi_vampire_policy.001.d

		ai_chance = {
			base = 10

			modifier = {
				add = 20
				has_trait = generous
			}

			modifier = {
				add = 20
				has_trait = patient
			}

			modifier = {
				add = -40
				has_trait = greedy
			}

			modifier = {
				add = 50
				gold > medium_gold_value
			}

			modifier = {
				gold < 100
				add = -30
			}

			ai_value_modifier = {
				ai_compassion = 0.5 
				ai_rationality = 0.5
			}
		}
		vi_vampire_county_investment_effect = yes
	}

	after = {
		vi_vampire_policy_after_effect = yes
	}

}
