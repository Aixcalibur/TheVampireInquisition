vi_investigate_secrets.001 = {

	title = "Success"
	desc = "I found out a horrid secret"

	type = character_event
	theme = intrigue
	left_portrait = {
		character = scope:target
		animation = personality_callous
	}

	right_portrait = {
		character = root
		animation = shock
	}

	immediate = {
		scope:target = {
			random_secret = {
				limit = {
					secret_type = secret_vi_vampire
					NOT = { is_known_by = scope:reveal_to_me }
				}

				reveal_to = root
				save_scope_as = vampire_secret
			}
		}

		add_opinion = {
			target = scope:target
			modifier = suspicion_opinion
			value = -20
		}
	}

	option = { #Expose secret
		name = vi_investigate.001.a

		scope:vampire_secret = {
			expose_secret = root
		}

		ai_chance = {
			base = 50
			ai_value_modifier = {
				ai_zeal = low_chance_impact_positive_ai_value
			}
			modifier = {
				has_trait = zealous
				add = 50
			}
			modifier = {
				opinion = {
					target = scope:owner
					value < 0
				}
				add = 100
			}
		}
	}

	option = {
		name = vi_investigate.001.b

		ai_chance = {
			base = 0

			modifier = {
				opinion = {
					target = scope:owner
					value > 0
				}
				add = 100
			}
		}
	}
}
