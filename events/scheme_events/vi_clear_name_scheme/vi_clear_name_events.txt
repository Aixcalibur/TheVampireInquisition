vi_clear_name.1001 = {
	hidden = yes

	immediate = {
		scope:scheme = {
			scheme_owner = {
				save_scope_as = owner
				#SUCCESS ROLL
				random = {
					chance = scope:scheme.scheme_success_chance
					save_scope_value_as = {
						name = scheme_successful
						value = yes
					}
				}
				if = {
					limit = { exists = scope:scheme_successful }
					trigger_event = {
						id = vi_clear_name.1002
						days = 1
					}
				}
				else = {
					trigger_event = {
						id = vi_clear_name.1003
						days = 1
					}	
				}
			}
		}
	}
}
