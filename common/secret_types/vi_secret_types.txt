
secret_vi_vampire = {
	category = murder

	is_valid = {
		scope:secret_owner = {
			vi_is_vampire_trigger = yes
		}
	}

	is_shunned = {
		always = no
	}

	is_criminal = {
		always = no
	}

	on_discover = {
		scope:discoverer = {
			if = {
				limit = {
					NOR = {
						vi_is_vampire_trigger = yes
						vi_is_vampire_accepting_trigger = yes
					}
				}

				add_opinion = {
					target = scope:secret_owner
					modifier = suspicion_opinion
					value = -20
				}
			}
		}
	}

	on_expose = {
		scope:secret_owner = {
			add_trait = vi_exposed_vampire
			# add_secret = {
			# 	type = secret_vi_vampire
			# }
		}

		secret_exposed_notification_effect = yes
	}
}
