vi_fateweaver_interaction = {
	icon = child
	common_interaction = yes
	category = interaction_category_admin
	interface_priority = 100
	desc = vi_fateweaver_interaction_desc

	cooldown = {
		years = 100
	}

	# cooldown_against_recipient = {
	# 	years = 5
	# }

	is_shown = {
		# scope:recipient = {
		# 	is_adult = yes #this is already part of the is available trigger
		# }
		scope:actor = {
			has_perk = vi_fateweaver_perk
			NOT = {
				this = scope:recipient
			}
			NOT = {
				owns_story_of_type = story_vi_fateweaver
			}
		}

		scope:recipient = {
			age < 5
		}
	}
	
	is_highlighted = {
		always = yes
	}

	is_valid_showing_failures_only = {
		scope:recipient = {
			NOT = {
				is_courtier_of = scope:actor
			}
		}
		scope:recipient = {
			health > 2
		}
		scope:recipient = {
			NOT = {
				has_relation_ward = scope:actor
			}
		}
		scope:recipient = {
			NOT = {
				liege = scope:actor
			}
		}
		NOT = {
			has_game_rule = no_conquerors_frequency
		}
	}

	cost = {
		piety = 100
	}
	
	on_accept = {
		scope:actor = {
			add_character_modifier = {
				modifier = vi_vampire_strain_modifier
				years = 50
			}
		}

		scope:recipient = {
			save_scope_as = destiny_child
			add_trait = lifestyle_mystic
			custom_tooltip = vi_fateweaver_description_tt
			hidden_effect = {
				add_character_flag = {
					flag = immune_to_disease
					years = 20
				}
			}
		}

		create_story = {
			type = story_vi_fateweaver
			save_scope_as = story
		}

		random_owned_story = {
			limit = {
				story_type = story_vi_fateweaver
			}
			set_variable = {
				name = destiny_child
				value = scope:destiny_child
			}
		}

		custom_tooltip = vi_fateweaver_warning_tt
	}
	
	auto_accept = yes

	ai_frequency = 1000

	ai_targets = {
		ai_recipients = guests
		ai_recipients = courtiers
	}

	ai_target_quick_trigger = {
		adult = no
	}

	ai_potential = {
		always = no
	}

	ai_will_do = {
		base = -100
	}
}
