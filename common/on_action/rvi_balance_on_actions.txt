on_birthday = {
	on_actions = {
		rvi_track_balance
	}
}

rvi_track_balance = {
	vi_is_vampire_trigger = yes
	effect = {
		if = {
			limit = {
				rvi_has_positive_perks = yes
				NOT = { has_character_flag = rvi_stop_trait_spam }
			}
			trigger_event = rvi_balance.0001
		}			
		else_if = {
			limit = {
				rvi_has_positive_perks = yes
				has_character_flag = rvi_stop_trait_spam
			}
			rvi_remove_all_positive_effects = yes		
		}
	}
}
