rvi_balance.0004 = {
	type = character_event
	title = rvi_balance.0004.title
	desc = rvi_balance.0004.desc

	theme = mental_health

	right_portrait = {
		character = root
		animation = dismissal
	}
	left_portrait = {
		character = scope:new_victim
		animation = rage
	}

	override_background = { reference = wilderness_activity }	

	immediate = {
		create_character = {
			template = criminal_thief_bandit_character
			save_scope_as = new_victim
		}
		hidden_effect = {
			scope:new_victim = {
				add_character_modifier = {
					modifier = malnourished_modifier
					years = 1
				}		
			}		
		}
	}

	option = {
		name = rvi_balance.0004.a1
		custom_tooltip = rvi_balance.0004.a1_tt
		hidden_effect = {
			scope:new_victim = {
				set_employer = root
			}		
		}	
		appoint_court_position = {
			recipient = scope:new_victim
			court_position = rvi_victim_court_position
		}			
		scope:new_victim = {
			add_trait = loyal
			add_opinion = {
				target = root
				modifier = friendliness_opinion
				opinion = 50
			}
		}
	}
	option = {
		name = rvi_balance.0004.a2
		custom_tooltip = rvi_balance.0004.a2_tt	
	}		
}
