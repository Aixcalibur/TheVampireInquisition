rvi_ritual.0002 = {
	type = activity_event

	}

			add_trait = loyal
		}

		}

			target = scope:inviter
			modifier = grateful_opinion

		}
	}

	option = {
		name = rvi_ritual.0002.b

			NOT = { has_trait = vi_heartstealer	}

			}		
		}
		appoint_court_position = {
			recipient = scope:created_food

		}			
		reverse_add_opinion = {
			target = scope:inviter
			modifier = grateful_opinion
			opinion = 15
		}
	}

	option = {
		name = rvi_ritual.0002.c

			}

		}
