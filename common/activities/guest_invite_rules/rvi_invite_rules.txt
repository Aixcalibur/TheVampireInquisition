activity_invite_rule_house_vampires = {
	effect = {
		house ?= {
			every_house_member = {
				limit = {
					vi_is_vampire_trigger = yes
					# Remove diarchs from the list.
					bannable_serving_diarch_trigger = no
				}
				add_to_list = characters
			}
		}
	}
}
