rvi_exalt_interaction = {
	icon = child
	common_interaction = yes
	category = interaction_category_friendly
	interface_priority = 100
	desc = rvi_exalt_interaction

	cooldown_against_recipient = {
		years = 10
 	}

	is_shown = {
		scope:recipient = {
			has_trait = vi_dhampir
			
		}
		scope:actor = {
			vi_is_vampire_trigger = yes
			dynasty = {
				has_dynasty_perk = rvi_one_legacy_1
				
			}
		}
	}
	
	is_highlighted = {
		always = yes
	}

	is_valid_showing_failures_only = {
		scope:recipient = {
			age >= 16
		}
	}

	cost = {
		piety = 200
	}
	
	on_accept = {
		scope:recipient = {
			add_opinion = {
				target = scope:actor
				modifier = grateful_opinion
				opinion = 10
			}
		
			
			save_scope_as = vi_target
			
			custom_tooltip = rvi_exalt_description_tt
			hidden_effect = {
				add_character_flag = {
					flag = immune_to_disease
					years = 20
				}
			}
		}
		
		scope:actor = {
			save_scope_as = vi_vampire
			trigger_event = rvi_legacies.1001
			add_character_modifier = {
				modifier = vi_vampire_strain_modifier
				years = 10
			}
		}
	}
	
	auto_accept = yes

	ai_frequency = 1000

	ai_targets = {
		ai_recipients = guests
		ai_recipients = courtiers
	}

	ai_target_quick_trigger = {
		adult = no
	}

	ai_potential = {
		always = no
	}

	ai_will_do = {
		base = -100
	}
}

				has_trait = beauty_good_3

		}
		flag = hook

	}
	send_options_exclusive = no

		scope:actor = { has_usable_hook = scope:recipient }

		piety = 800

				modifier = rvi_ame_comblee_modifier
			}			
		}
		scope:actor = {
			stress_impact = {
				base = medium_stress_impact_gain
			}

				modifier = vi_vampire_strain_modifier
				years = 10
			}

				modifier = vi_drained_modifier

			}

	}

					limit = {
						secret_type = secret_vi_vampire

				}
				add_opinion = {
					target = scope:recipient
					modifier = disappointed_opinion

				}
			}
		}
	}

 		base = -50

 		opinion_modifier = {
 			opinion_target = scope:actor

 		}

			}
			AND = {
