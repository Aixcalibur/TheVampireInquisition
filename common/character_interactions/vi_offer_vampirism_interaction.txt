	ai_frequency = 60
	
	is_shown = {
		NOT = {
			scope:actor = scope:recipient

			vi_is_vampire_trigger = yes
			is_ruler = yes
		}
		scope:recipient = {
			vi_is_vampire_trigger = no

			OR = {
				is_courtier_of = scope:actor
				is_ruler = yes
			}
		}
	}

	is_valid_showing_failures_only = {
		
		scope:recipient = {
			custom_tooltip = {
				is_landed = yes
				OR = {
					has_relation_friend = scope:actor
					has_relation_lover = scope:actor
					has_relation_soulmate = scope:actor
					has_relation_best_friend = scope:actor
				}
				text = VI_MUST_HAVE_RELATION_TO_OFFER
			}
		}
			
		

		scope:recipient = {
			NOT = {
				has_trait = vi_dhampir
			}
		}
	}

	cost = {
		piety = 100
	}

	on_accept = {
		
		scope:actor = {
			add_character_modifier = {
				modifier = vi_vampire_strain_modifier
				years = 5
			}
			stress_impact = {
				base = medium_stress_impact_gain
			}
			save_scope_as = vi_vampire
		}

		scope:recipient = {
			add_opinion = {
				target = scope:actor
				modifier = grateful_opinion
				opinion = 20
			}

			save_scope_as = vi_target

			vi_turn_into_vampire_effect = yes

			if = {
				limit = {
					is_ai = no
				}
				add_trait = vi_vampire_1
			}
		}
	}

	on_decline = {
		scope:actor = {
			random_secret = {
				limit = {
					secret_type = secret_vi_vampire
				}
				reveal_to = scope:recipient
			}
			add_opinion = {
				target = scope:recipient
				modifier = disappointed_opinion
				opinion = -60
			}
		}
	}

	ai_accept = {
 		base = -50

 		opinion_modifier = {
 			opinion_target = scope:actor

 			step = 5
