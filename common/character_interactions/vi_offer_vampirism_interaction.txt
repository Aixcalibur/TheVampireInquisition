vi_offer_vampirism_interaction = {
	category = interaction_category_friendly

	desc = vi_offer_vampirism_interaction_desc

	ai_maybe = yes

	on_decline_summary = vi_offer_vampirism_decline_summary

	cooldown_against_recipient = {
		years = 10

	ai_targets = {
		ai_recipients = liege

	ai_targets = {
		ai_recipients = neighboring_rulers
		ai_recipients = peer_vassals
		max = 10
	}

	ai_frequency = 60
	
	is_shown = {
		NOT = {
			scope:actor = scope:recipient
		}
		scope:actor = {
			vi_is_vampire_trigger = yes
			is_ruler = yes
		}
		scope:recipient = {
			vi_is_vampire_trigger = no

			OR = {
				is_courtier_of = scope:actor
				is_ruler = yes
			}
		}
	}

	is_valid_showing_failures_only = {
		
		scope:recipient = {
			NOT = {
				has_trait = vi_dhampir
			}
			
			trigger_if = {
				limit = {
					is_landed = yes
					is_ruler = yes
				}
				custom_tooltip = {
					OR = {
						has_relation_friend = scope:actor
						has_relation_lover = scope:actor
						has_relation_soulmate = scope:actor
						has_relation_best_friend = scope:actor
					}
					text = VI_MUST_HAVE_RELATION_TO_OFFER
				}
			}
		}
	}

	cost = {
		piety = 100
	}

	send_option = { 
		is_shown = { 
			scope:actor = {
				exists = house
			}
			scope:recipient = {
				house ?= {
					NOT = {
						this = scope:actor.house
					}
				}
			}
		}
		can_be_changed = {
			always = yes
		}
		is_valid = {
			scope:actor = {
				#exists = house
				house ?= {
					house_head = scope:actor
				}
			}
		}
		flag = adoption_send_option
		localization = VI_ASK_TO_BE_ADOPTED
	}

	send_options_exclusive = no

	on_accept = {
		
		scope:actor = {
			add_character_modifier = {
				modifier = vi_vampire_strain_modifier
				years = 5
			}
			stress_impact = {
				base = medium_stress_impact_gain
			}
			save_scope_as = vi_vampire
		}

		scope:recipient = {
			add_opinion = {
				target = scope:actor
				modifier = grateful_opinion
				opinion = 20
			}

			save_scope_as = vi_target

			vi_turn_into_vampire_effect = yes

			if = {
				limit = {
					scope:actor = {
						is_ai = no
					}
				}
				add_trait = vi_vampire_1
			}

			if = {
				limit = {
					OR = {
						NOT = {
							exists = house
						}
						always = scope:adoption_send_option
					}
				}
				set_house = scope:actor.house
			}
		}
	}

	on_decline = {
		scope:actor = {
			send_interface_toast = {
				title = vi_vampirism_declined_msg
				right_icon = scope:recipient
				random_secret = {
					limit = {
						secret_type = secret_vi_vampire
					}
					reveal_to = scope:recipient
				}
				add_opinion = {
					target = scope:recipient
					modifier = disappointed_opinion
					opinion = -60
				}
			}
		}
	}

	ai_accept = {
 		base = -25

 		opinion_modifier = {
 			opinion_target = scope:actor
 			multiplier = 0.5
 			step = 5
 		}

		modifier = {
			add = {
				value = scope:recipient.dynasty.dynasty_prestige_level
				multiply = -10
			}
			AND = {
				always = scope:adoption_send_option
				scope:recipient = {
					exists = dynasty
					dynasty ?= {
						any_dynasty_member = {
							count >= 3
						}
						dynasty_prestige_level >= 2
					}
				}
			}
			
			desc = VI_ASKING_TO_BE_ADOPTED
		}

 		#Relations & relatives (with actor)
 		modifier = {

				has_relation_soulmate = scope:actor
			}

 		}

				}
			}

			has_trait = vi_inquisitor
			add = -50
		}

		modifier = {
			OR = {
				has_trait = sadistic
				has_trait = callous
				ai_compassion < -50
			}
			add = 25
			desc = VI_DOESNT_MIND_BLOOD
		}

		modifier = {
			OR = {
				has_trait = compassionate
				ai_compassion > 50
			}
			add = -25
			desc = VI_MINDS_BLOOD
		}

		modifier = {
			NOR = {
				has_trait = zealous
				has_trait = humble
				has_trait = content
			}
			OR = {
				has_trait = ambitious
				AND = {
					ai_greed > 50
					ai_boldness > 30
				}
			}
			desc = VI_WANTS_TO_LIVE
			add = 100
		}
