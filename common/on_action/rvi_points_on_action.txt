on_birthday = {
	on_actions = {
		rvi_track_points
	}
}

rvi_track_points = {
	vi_is_vampire_trigger = yes
	effect = {
		if = {
			limit = {
				rvi_has_freepoints = yes
				NOT = { has_character_flag = rvi_stop_skill_spam }
			}
			trigger_event = rvi_balance.0002
		}		
		else_if = {
			limit = {
				rvi_has_freepoints = yes
				has_character_flag = rvi_stop_skill_spam
			}
			rvi_remove_all_skillpoints = yes			
		}
	}
}
